% Define a command for mutliple blank lines
\newcounter{blanklinescounter} % needed by the blanklines command
\newcommand{\blanklines}[1]{%
  \setcounter{blanklinescounter}{0}
  \forloop{blanklinescounter}{1}{\value{blanklinescounter} < #1}{%
    \hrulefill \\
  }%
}
